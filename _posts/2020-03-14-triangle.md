```c++
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

int dp[501][501];

int main()
{
	int n, sum = -1;

	cin >> n;

	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= i; j++)
		{
			cin >> dp[i][j];
		}
	}

	for (int i = 2; i <= n; ++i)
	{
		for (int j = 1; j <= i; ++j)
		{
			if (j == 1)
			{
				dp[i][j]+=dp[i-1][j];
			}
			else if (j == n)
			{
				dp[i][j] += dp[i - 1][j - 1];
			}
			else
			{
				dp[i][j] += max(dp[i - 1][j], dp[i - 1][j - 1]);
			}
		}
	}

	for (int i = 1; i <= n; i++)
	{
		sum = max(sum, dp[n][i]);
	}

	cout << sum;
	return 0;
}
```
